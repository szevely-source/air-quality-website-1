      </div>

      <div class="grid" style="align-items:start">
        <!-- Explanations -->
        <div class="card" style="grid-column: span 2;">
          <h3>Key Metrics and Instruments</h3>
          <p class="muted">Air quality monitoring uses a combination of fixed regulatory stations, mobile sensors, satellite retrievals, and low-cost community sensors to measure chemical species and particle concentrations.</p>

          <ul style="margin-top:12px">
            <li><strong>PM2.5 / PM10:</strong> Fine and coarse particulate matter — measured in µg/m³ — strongly linked to health outcomes.</li>
            <li><strong>Ozone (O₃):</strong> A secondary pollutant that causes respiratory irritation and is measured in ppb (parts per billion).</li>
            <li><strong>NO₂, SO₂, CO:</strong> Gas-phase pollutants monitored by chemiluminescence, UV fluorescence, and electrochemical sensors.</li>
            <li><strong>Air Quality Index (AQI):</strong> A standardized scale (0–500) that translates pollutant concentrations into health-based categories like Good, Moderate, Unhealthy.</li>
          </ul>

          <p class="muted small">Regulatory monitors follow strict quality-assurance procedures; low-cost sensors supplement coverage but require calibration and QA for scientific or policy use.</p>
        </div>

      const sampleAQI = [95, 88, 85, 82, 78, 76, 73, 70, 67]; // illustrative downward trend

          }

    /* ---------------------------
       3) Live measurements widget using OpenAQ (no API key)
       - This is optional: user can enter a city name to fetch 'latest' measurements.
       - OpenAQ returns stations and measurements; we pick the first result for illustration.
       ---------------------------- */
    (function setupAQIWidget(){
      const btn = document.getElementById('searchBtn');
      const exampleBtn = document.getElementById('exampleBtn');
      const input = document.getElementById('cityInput');

      async function fetchCityMeasurements(city){
        const endpoint = `https://api.openaq.org/v2/latest?city=${encodeURIComponent(city)}&limit=5`;
        try{
          const res = await fetch(endpoint);
          if(!res.ok) throw new Error('Network response not ok');
          const data = await res.json();
          return data;
        }catch(err){
          console.error(err);
          return null;
        }
      }

      function displayResult(data){
        const resultBox = document.getElementById('aqiResult');
        const locEl = document.getElementById('aq-location');
        const updatedEl = document.getElementById('aq-updated');
        const pm25El = document.getElementById('aq-pm25');
        const pm10El = document.getElementById('aq-pm10');
        const srcEl = document.getElementById('aq-source');

        if(!data || !data.results || data.results.length === 0){
          alert('No recent measurements found for that city. Try another name or check spelling.');
          return;
        }

        // pick the first station result for demonstration
        const station = data.results[0];
        const cityName = station.city || station.location || 'Unknown location';
        const measurements = station.measurements || [];

        // try to find PM2.5 and PM10
        const pm25 = measurements.find(m => m.parameter.toLowerCase() === 'pm25');
        const pm10 = measurements.find(m => m.parameter.toLowerCase() === 'pm10');

        locEl.textContent = `${cityName} — ${station.location}`;
        updatedEl.textContent = 'Updated: ' + (measurements[0] ? new Date(measurements[0].lastUpdated).toLocaleString() : '—');
        pm25El.textContent = pm25 ? `${pm25.value} ${pm25.unit}` : 'N/A';
        pm10El.textContent = pm10 ? `${pm10.value} ${pm10.unit}` : 'N/A';
        srcEl.textContent = station.sourceName || 'OpenAQ';

        resultBox.style.display = 'block';
      }

      btn.addEventListener('click', async () => {
        const city = input.value.trim();
        if(!city){ alert('Please enter a city name.'); input.focus(); return; }
        btn.disabled = true;
        btn.textContent = 'Loading...';
        const data = await fetchCityMeasurements(city);
        btn.textContent = 'Get latest';
        btn.disabled = false;
        displayResult(data);
      });

      // convenience button to show example data
      exampleBtn.addEventListener('click', async () => {
        input.value = 'Los Angeles';
        exampleBtn.disabled = true;
        exampleBtn.textContent = 'Loading...';
        const data = await fetchCityMeasurements('Los Angeles');
        exampleBtn.textContent = 'Example: Los Angeles';
        exampleBtn.disabled = false;
        displayResult(data);
      });
    })();
  </script>
